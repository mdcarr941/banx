     { $match: { "tags.key": "Ans" } },
     { $project: { tags: { $filter: {
          input: "$tags",
          as: "tag",
          cond: { $eq: ["$$tag.key", "Ans"] }
     }}}},
     { $project: { values: { $map: {
          input: "$tags",
          as: "tag",
          in: "$$tag.value"
     }}}},
     { $group: { _id: null, values: { $addToSet: "$values" }}},
     { $project: { _id: 0, values: { $reduce: {
          input: "$values",
          initialValue: [],
          in: { $concatArrays: ["$$value", "$$this"] }
     }}}}
])
